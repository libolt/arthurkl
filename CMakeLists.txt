cmake_minimum_required(VERSION 2.8)

project(arthurkl)
SET(CMAKE_MODULE_PATH "${arthurkl_SOURCE_DIR}/cmake")
find_package (SDL2)
find_package(SDL_Image)
find_package (SDL_Mixer)
message(SDL2_INClUDE DIR = ${SDL2_INCLUDE_DIR})
INCLUDE_DIRECTORIES(${SDL2_INCLUDE_DIR} ${SDL_Image_INCLUDE_DIRS} ${SDL_Mixer_INCLUDE_DIRS})
LINK_DIRECTORIES()
message (SDL2_LIBRARY = ${SDL2_LIBRARY})
message (SDL_Image_LIBRARIES = ${SDL_Image_LIBRARIES})
add_executable(${PROJECT_NAME} "main.cpp")
TARGET_LINK_LIBRARIES(${PROJECT_NAME} ${SDL2_LIBRARY} ${SDL_Image_LIBRARIES} ${SDL_Mixer_LIBRARIES})
